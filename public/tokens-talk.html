    const bf16* x, const bf16* bias,
    float scale, bf16* y) {
  int i = blockIdx.x*blockDim.x + threadIdx.x;
  float v = float(x[i]) + float(bias[i%C]);
  v = __saturatef(v);              // clamp [0,1]
  v = __saturatef(v*scale) / scale;
  y[i] = __float2bfloat16(v);      // one pass
}
